# Script para subir el proyecto a GitHub
# Ejecuta estos comandos uno por uno en PowerShell

# 1. Inicializar repositorio Git
git init

# 2. Agregar el repositorio remoto
git remote add origin https://github.com/sjmontano/portfolio.git

# 3. Agregar todos los archivos
git add .

# 4. Hacer el primer commit
git commit -m "üéâ Initial commit: Portfolio personal completo con optimizaciones para GitHub Pages"

# 5. Crear y cambiar a la rama main (si no existe)
git branch -M main

# 6. Subir al repositorio
git push -u origin main

# 4. Haz clic en Save
